    const amountInput = document.getElementById('investAmount');
    const rateInput = document.getElementById('returnRate');
    const roiSpan = document.getElementById('roiResult');
    function updateROI() {
        let amt = parseFloat(amountInput.value) || 0;
        let rate = parseFloat(rateInput.value) || 0;
        let monthly = (amt * rate) / 100;
        roiSpan.innerText = '?' + monthly.toLocaleString('en-IN', { maximumFractionDigits: 0 });
    }
    amountInput.addEventListener('input', updateROI);
    rateInput.addEventListener('input', updateROI);
    updateROI();
